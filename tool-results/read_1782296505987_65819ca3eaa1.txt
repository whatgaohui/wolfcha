   102→// Flatten multipart content to plain string for models that don't support it
   103→function flattenMultipartContent(messages: unknown[]): unknown[] {
   104→  if (!Array.isArray(messages)) return messages;
   105→
   106→  return messages.map((msg) => {
   107→    if (!msg || typeof msg !== "object") return msg;
   108→    const m = msg as Record<string, unknown>;
   109→
   110→    // If content is an array, flatten to string
   111→    if (Array.isArray(m.content)) {
   112→      const textParts = m.content
   113→        .filter((part): part is { type: string; text: string } =>
   114→          part && typeof part === "object" && (part as { type?: string }).type === "text"
   115→        )
   116→        .map((part) => part.text || "")
   117→        .filter(Boolean);
   118→      
   119→      return { ...m, content: textParts.join("\n\n") };
   120→    }
   121→
   122→    return m;
   123→  });
   124→}

   126→// DeepSeek context caching is prefix-based. Text-only multipart messages are
   127→// normalized to one plain string so identical text starts at token 0 reliably.
   128→function coalesceTextOnlyMultipartContent(messages: unknown[]): unknown[] {
   129→  if (!Array.isArray(messages)) return messages;
   130→
   131→  return messages.map((msg) => {
   132→    if (!msg || typeof msg !== "object") return msg;
   133→    const m = msg as Record<string, unknown>;
   134→
   135→    if (!Array.isArray(m.content)) return m;
   136→
   137→    const parts = m.content;
   138→    const allText = parts.every(
   139→      (part): part is { type: string; text: string } =>
   140→        part &&
   141→        typeof part === "object" &&
   142→        (part as { type?: string }).type === "text" &&
   143→        typeof (part as { text?: unknown }).text === "string"
   144→    );
   145→
   146→    if (!allText) return stripCacheControl([m])[0] as Record<string, unknown>;
   147→
   148→    return {
   149→      ...m,
   150→      content: parts
   151→        .map((part) => part.text.trim())
   152→        .filter(Boolean)
   153→        .join("\n\n"),
   154→    };
   155→  });
   156→}

   158→function prependDeepSeekStablePrefix(messages: unknown[]): unknown[] {
   159→  if (!Array.isArray(messages)) return messages;
   160→  let prepended = false;
   161→  const next = messages.map((msg) => {
   162→    if (prepended || !msg || typeof msg !== "object") return msg;
   163→    const m = msg as Record<string, unknown>;
   164→    if (m.role !== "system" || typeof m.content !== "string") return m;
   165→    prepended = true;
   166→    if (m.content.includes(DEEPSEEK_STABLE_PREFIX_MARKER)) return m;
   167→    return {
   168→      ...m,
   169→      content: `${DEEPSEEK_STABLE_PROMPT_CACHE_PREFIX}\n\n${m.content}`,
   170→    };
   171→  });
   172→
   173→  if (prepended) return next;
   174→  return [
   175→    { role: "system", content: DEEPSEEK_STABLE_PROMPT_CACHE_PREFIX },
   176→    ...next,
   177→  ];
   178→}

   180→function hasJsonHintInMessages(messages: unknown[]): boolean {
   181→  if (!Array.isArray(messages)) return false;
   182→
   183→  const contains = (value: unknown): boolean => {
   184→    if (typeof value === "string") return /json/i.test(value);
   185→    if (Array.isArray(value)) return value.some(contains);
   186→    if (!value || typeof value !== "object") return false;
   187→    const obj = value as Record<string, unknown>;
   188→    if ("text" in obj && typeof obj.text === "string") return /json/i.test(obj.text);
   189→    if ("content" in obj) return contains(obj.content);
   190→    return false;
   191→  };
   192→
   193→  return messages.some((m) => contains(m));
   194→}

   202→// Strip cache_control from message content parts for models that don't support it
   203→function stripCacheControl(messages: unknown[]): unknown[] {
   204→  if (!Array.isArray(messages)) return messages;
   205→
   206→  return messages.map((msg) => {
   207→    if (!msg || typeof msg !== "object") return msg;
   208→    const m = msg as Record<string, unknown>;
   209→
   210→    // If content is an array (multipart), strip cache_control from each part
   211→    if (Array.isArray(m.content)) {
   212→      const strippedContent = m.content.map((part) => {
   213→        if (part && typeof part === "object" && "cache_control" in part) {
   214→          // eslint-disable-next-line @typescript-eslint/no-unused-vars
   215→          const { cache_control, ...rest } = part as Record<string, unknown>;
   216→          return rest;
   217→        }
   218→        return part;
   219→      });
   220→      return { ...m, content: strippedContent };
   221→    }
   222→
   223→    return m;
   224→  });
   225→}

   227→// ZenMux reasoning: only enabled, effort, max_tokens (see docs.zenmux.ai/guide/advanced/reasoning.html)

   237→function isReasoningEffort(value: unknown): value is ReasoningEffort {

   241→/** Build ZenMux request reasoning object (no unsupported fields like exclude). */
   242→function toZenMuxReasoning(
   243→  r: { enabled?: boolean; effort?: string; max_tokens?: number } | undefined
   244→): { enabled: boolean; effort?: string; max_tokens?: number } {
   245→  if (r?.enabled === true) {
   246→    return {
   247→      enabled: true,
   248→      ...(r.effort != null && { effort: r.effort }),
   249→      ...(typeof r.max_tokens === "number" && Number.isFinite(r.max_tokens) && { max_tokens: r.max_tokens }),
   250→    };
   251→  }
   252→  return { enabled: false };
   253→}

   255→function toTokendanceThinking(
   256→  r: { enabled?: boolean; effort?: string; max_tokens?: number } | undefined
   257→): Record<string, unknown> | undefined {
   258→  if (r === undefined) return undefined;
   259→  if (r.enabled !== true) return { type: "disabled" };
   260→
   261→  const effortBudget: Record<string, number> = {
   262→    minimal: 64,
   263→    low: 128,
   264→    medium: 256,
   265→    high: 512,
   266→  };
   267→  const budget =
   268→    typeof r.max_tokens === "number" && Number.isFinite(r.max_tokens)
   269→      ? Math.max(32, Math.floor(r.max_tokens))
   270→      : r.effort
   271→        ? effortBudget[r.effort]
   272→        : undefined;
   273→
   274→  return {
   275→    type: "enabled",
   276→    ...(budget ? { budget_tokens: budget } : {}),
   277→  };
   278→}

   292→async function runBatchItem(
   293→  payload: ChatRequestPayload,
   294→  headerApiKey: string | null,
   295→  headerDashscopeKey: string | null,
   296→  headerTokendanceKey: string | null,
   297→  headerTokendanceBaseUrl: string | null
   298→): Promise<{ ok: true; data: unknown } | { ok: false; status: number; error: string; details?: unknown }> {
   299→  const {
   300→    model,
   301→    messages,
   302→    temperature,
   303→    max_tokens,
   304→    stream,
   305→    reasoning,
   306→    reasoning_effort,
   307→    response_format,
   308→    provider,
   309→  } = payload;
   310→
   311→  if (stream) {
   312→    return { ok: false, status: 400, error: "Batch request does not support stream=true" };
   313→  }
   314→
   315→  const modelProvider: Provider | null =
   316→    provider === "dashscope" || provider === "zenmux" || provider === "tokendance" ? provider : getProviderForModel(model);
   317→  if (!modelProvider) {
   318→    return { ok: false, status: 400, error: `Unknown model: ${String(model ?? "").trim() || "unknown"}` };
   319→  }
   320→
   321→  const isDefaultModel = PROJECT_MODELS.some((ref) => ref.model === model);
   322→  if (!isDefaultModel) {
   323→    if (modelProvider === "zenmux" && !headerApiKey) {
   324→      return { ok: false, status: 401, error: "此模型需要您提供 Zenmux API Key" };
   325→    }
   326→    if (modelProvider === "dashscope" && !headerDashscopeKey) {
   327→      return { ok: false, status: 401, error: "此模型需要您提供百炼 API Key" };
   328→    }
   329→    if (modelProvider === "tokendance" && !headerTokendanceKey) {
   330→      return { ok: false, status: 401, error: "此模型需要您提供 TokenDance Key" };
   331→    }
   332→  }
   333→
   334→  const hasAnyCustomKeyHeader = Boolean((headerApiKey ?? "").trim() || (headerDashscopeKey ?? "").trim() || (headerTokendanceKey ?? "").trim());
   335→
   336→  const modelRefOverride = getModelRef(model);
   337→  const normalizedTemperature =
   338→    modelRefOverride?.temperature !== undefined
   339→      ? modelRefOverride.temperature
   340→      : (typeof temperature === "number" && Number.isFinite(temperature) ? temperature : 0.7);
   341→  const cappedTemperature = (() => {
   342→    const lower = typeof model === "string" ? model.toLowerCase() : "";
   343→    const needZeroOne =
   344→      modelProvider === "zenmux" ||
   345→      lower.startsWith("moonshotai/") ||
   346→      lower.includes("kimi");
   347→    if (needZeroOne) {
   348→      return Math.min(Math.max(0, normalizedTemperature), 1);
   349→    }
   350→    return Math.max(0, normalizedTemperature);
   351→  })();
   352→  const effectiveReasoning = modelRefOverride?.reasoning !== undefined ? modelRefOverride.reasoning : reasoning;
   353→
   354→  let processedMessages: unknown[] = messages;
   355→  if (!supportsMultipartContent(model)) {
   356→    processedMessages = flattenMultipartContent(processedMessages);
   357→  } else if (supportsAutomaticPrefixCaching(model)) {
   358→    processedMessages = prependDeepSeekStablePrefix(coalesceTextOnlyMultipartContent(processedMessages));
   359→  } else if (modelProvider === "dashscope") {
   360→    processedMessages = stripCacheControl(processedMessages);
   361→  } else if (!supportsExplicitCaching(model)) {
   362→    processedMessages = stripCacheControl(processedMessages);
   363→  }
   364→
   365→  if (modelProvider === "dashscope") {
   366→    if (hasAnyCustomKeyHeader && !headerDashscopeKey) {
   367→      return { ok: false, status: 401, error: "已启用自定义 Key，但未提供百炼 API Key（已拒绝回退到系统 Key）" };
   368→    }
   369→    const dashscopeApiKey = headerDashscopeKey || process.env.DASHSCOPE_API_KEY;
   370→    if (!dashscopeApiKey) {
   371→      return { ok: false, status: 500, error: "DASHSCOPE_API_KEY not configured on server" };
   372→    }
   373→
   374→    const normalizedModel = normalizeDashscopeModelName(model);
   375→    const normalizedResponseFormat = response_format as { type?: unknown } | undefined;
   376→    const dashscopeMessages =
   377→      normalizedResponseFormat?.type === "json_object"
   378→        ? withDashscopeJsonHint(processedMessages)
   379→        : processedMessages;
   380→
   381→    const requestBody: Record<string, unknown> = {
   382→      model: normalizedModel,
   383→      messages: dashscopeMessages,
   384→      temperature: cappedTemperature,
   385→    };
   386→
   387→    if (typeof max_tokens === "number" && Number.isFinite(max_tokens)) {
   388→      requestBody.max_tokens = Math.max(16, Math.floor(max_tokens));
   389→    }
   390→
   391→    if (response_format) {
   392→      requestBody.response_format = response_format;
   393→    }
   394→
   395→    const controller = new AbortController();
   396→    const timeoutId = setTimeout(() => controller.abort(), API_TIMEOUT_MS);
   397→
   398→    let response: Response;
   399→    try {
   400→      response = await fetch(DASHSCOPE_CHAT_COMPLETIONS_URL, {
   401→        method: "POST",
   402→        headers: {
   403→          Authorization: `Bearer ${dashscopeApiKey}`,
   404→          "Content-Type": "application/json",
   405→        },
   406→        body: JSON.stringify(requestBody),
   407→        signal: controller.signal,
   408→      });
   409→    } finally {
   410→      clearTimeout(timeoutId);
   411→    }
   412→
   413→    if (!response.ok) {
   414→      const errorText = await response.text();
   415→      let parsed: unknown = undefined;
   416→      try {
   417→        parsed = JSON.parse(errorText);
   418→      } catch {
   419→        // ignore
   420→      }
   421→      return {
   422→        ok: false,
   423→        status: response.status,
   424→        error: `DashScope API error: ${response.status}`,
   425→        details: parsed ?? errorText,
   426→      };
   427→    }
   428→
   429→    const result = await response.json();
   430→    return { ok: true, data: result };
   431→  }
   432→
   433→  if (modelProvider === "tokendance") {
   434→    if (hasAnyCustomKeyHeader && !headerTokendanceKey) {
   435→      return { ok: false, status: 401, error: "已启用自定义 Key，但未提供 TokenDance Key（已拒绝回退到系统 Key）" };
   436→    }
   437→    const tokendanceApiKey = headerTokendanceKey || process.env.TOKENDANCE_API_KEY;
   438→    const tokendanceBaseUrl = headerTokendanceBaseUrl || process.env.TOKENDANCE_BASE_URL || TOKENDANCE_BASE_URL;
   439→    if (!tokendanceApiKey || !tokendanceBaseUrl) {
   440→      return { ok: false, status: 500, error: "TOKENDANCE_API_KEY or TOKENDANCE_BASE_URL not configured on server" };
   441→    }
   442→
   443→    const tokendanceUrl = getTokendanceUrl(tokendanceBaseUrl);
   444→    if (!tokendanceUrl) {
   445→      return { ok: false, status: 500, error: "Invalid TokenDance Base URL" };
   446→    }
   447→
   448→    const requestBody: Record<string, unknown> = {
   449→      model,
   450→      messages: processedMessages,
   451→      temperature: cappedTemperature,
   452→    };
   453→
   454→    if (typeof max_tokens === "number" && Number.isFinite(max_tokens)) {
   455→      requestBody.max_tokens = Math.max(16, Math.floor(max_tokens));
   456→    }
   457→
   458→    // GLM-4.7 / Kimi K2.5 默认开启思考，API 参数可关闭（已实测有效）
   459→    const modelLower = model.toLowerCase();
   460→    const thinking = toTokendanceThinking(effectiveReasoning);
   461→    if (thinking) {
   462→      requestBody.thinking = thinking;
   463→    } else if (modelLower.includes("glm") || modelLower.includes("kimi")) {
   464→      requestBody.thinking = { type: "disabled" };
   465→    }
   466→
   467→    if (response_format && supportsResponseFormat(model)) {
   468→      requestBody.response_format = response_format;
   469→    }
   470→
   471→    const controller = new AbortController();
   472→    const timeoutId = setTimeout(() => controller.abort(), API_TIMEOUT_MS);
   473→
   474→    let response: Response;
   475→    try {
   476→      response = await fetch(tokendanceUrl, {
   477→        method: "POST",
   478→        headers: {
   479→          Authorization: `Bearer ${tokendanceApiKey}`,
   480→          "Content-Type": "application/json",
   481→          "X-App-Name": "Wolfcha",
   482→          "X-Site-URL": "https://wolf-cha.com",
   483→        },
   484→        body: JSON.stringify(requestBody),
   485→        signal: controller.signal,
   486→      });
   487→    } finally {
   488→      clearTimeout(timeoutId);
   489→    }
   490→
   491→    if (!response.ok) {
   492→      const errorText = await response.text();
   493→      let parsed: unknown = undefined;
   494→      try {
   495→        parsed = JSON.parse(errorText);
   496→      } catch {
   497→        // ignore
   498→      }
   499→      return {
   500→        ok: false,
   501→        status: response.status,
   502→        error: `TokenDance error: ${response.status}`,
   503→        details: parsed ?? errorText,
   504→      };
   505→    }
   506→
   507→    const result = await response.json();
   508→    return { ok: true, data: result };
   509→  }
   510→
   511→  if (hasAnyCustomKeyHeader && !headerApiKey) {
   512→    return { ok: false, status: 401, error: "已启用自定义 Key，但未提供 Zenmux API Key（已拒绝回退到系统 Key）" };
   513→  }
   514→
   515→  const apiKey = headerApiKey || process.env.ZENMUX_API_KEY;
   516→  if (!apiKey) {
   517→    return { ok: false, status: 500, error: "ZENMUX_API_KEY not configured on server" };
   518→  }
   519→
   520→  const requestBody: Record<string, unknown> = {
   521→    model,
   522→    messages: processedMessages,
   523→    temperature: cappedTemperature,
   524→  };
   525→
   526→  if (typeof max_tokens === "number" && Number.isFinite(max_tokens)) {
   527→    requestBody.max_tokens = Math.max(16, Math.floor(max_tokens));
   528→  }
   529→
   530→  const reasoningEffort = isReasoningEffort(reasoning_effort) ? reasoning_effort : undefined;
   531→  const reasoningToUse = effectiveReasoning ?? reasoning;
   532→  if (reasoningToUse !== undefined) {
   533→    requestBody.reasoning = toZenMuxReasoning(reasoningToUse);
   534→  } else if (reasoningEffort) {
   535→    requestBody.reasoning_effort = reasoningEffort;
   536→  } else {
   537→    requestBody.reasoning = { enabled: false };
   538→  }
   539→
   540→  if (response_format && supportsResponseFormat(model)) {
   541→    requestBody.response_format = response_format;
   542→  }
   543→
   544→  const controller = new AbortController();
   545→  const timeoutId = setTimeout(() => controller.abort(), API_TIMEOUT_MS);
   546→
   547→  let response: Response;
   548→  try {
   549→    response = await fetch(ZENMUX_API_URL, {
   550→      method: "POST",
   551→      headers: {
   552→        Authorization: `Bearer ${apiKey}`,
   553→        "Content-Type": "application/json",
   554→      },
   555→      body: JSON.stringify(requestBody),
   556→      signal: controller.signal,
   557→    });
   558→  } finally {
   559→    clearTimeout(timeoutId);
   560→  }
   561→
   562→  if (!response.ok) {
   563→    const errorText = await response.text();
   564→    return {
   565→      ok: false,
   566→      status: response.status,
   567→      error: `ZenMux API error: ${response.status} - ${errorText}`,
   568→    };
   569→  }
   570→
   571→  const result = await response.json();
   572→  return { ok: true, data: result };
   573→}
