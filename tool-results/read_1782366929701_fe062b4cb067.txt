    51→function shuffleArray<T>(array: T[]): T[] {
    52→  const shuffled = [...array];
    53→  for (let i = shuffled.length - 1; i > 0; i--) {
    54→    const j = Math.floor(Math.random() * (i + 1));
    55→    [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
    56→  }
    57→  return shuffled;
    58→}

    68→export const sampleModelRefs = (count: number): ModelRef[] => {
    69→  // Default pool when custom key is not enabled
    70→  const defaultPool =
    71→    PLAYER_MODELS.length > 0
    72→      ? PLAYER_MODELS
    73→      : [getModelRefForModel(GENERATOR_MODEL)];
    74→
    75→  const pool = (() => {
    76→    if (!isCustomKeyEnabled()) return defaultPool;
    77→
    78→    // When custom key is enabled, use ALL_MODELS as the full available pool
    79→    const fullPool = ALL_MODELS.length > 0 ? ALL_MODELS : defaultPool;
    80→
    81→    const allowedProviders = new Set<ModelRef["provider"]>();
    82→    allowedProviders.add("zai");
    83→    if (allowedProviders.size === 0) return defaultPool;
    84→
    85→    // Filter by allowed providers, then exclude non-player models
    86→    const allowedPool = filterPlayerModels(
    87→      fullPool.filter((ref) => allowedProviders.has(ref.provider))
    88→    );
    89→    if (allowedPool.length === 0) return defaultPool;
    90→
    91→    // Filter by user's selected models - STRICTLY respect user selection
    92→    const selectedModels = getSelectedModels();
    93→    if (selectedModels.length === 0) return allowedPool;
    94→    
    95→    // Only use models the user explicitly selected
    96→    const selectedPool = allowedPool.filter((ref) => selectedModels.includes(ref.model));
    97→    
    98→    // If user selected models but none are in allowedPool, try to find them in fullPool
    99→    // This handles cases where user selected models from a different provider
   100→    if (selectedPool.length === 0) {
   101→      const fullSelectedPool = filterPlayerModels(
   102→        fullPool.filter((ref) => selectedModels.includes(ref.model) && allowedProviders.has(ref.provider))
   103→      );
   104→      if (fullSelectedPool.length > 0) return fullSelectedPool;
   105→      
   106→      // Last resort: only return models that user actually selected, even if empty
   107→      // This prevents using models the user didn't choose
   108→      console.warn("[sampleModelRefs] User selected models not found in allowed pool:", selectedModels);
   109→    }
   110→    
   111→    // Return only user-selected models, never fall back to all models
   112→    return selectedPool.length > 0 ? selectedPool : allowedPool.slice(0, 1);
   113→  })();
   114→
   115→  if (!Number.isFinite(count) || count <= 0) return [];
   116→
   117→  if (count <= pool.length) {
   118→    return shuffleArray(pool).slice(0, count);
   119→  }
   120→
   121→  const out = shuffleArray(pool);
   122→  while (out.length < count) {
   123→    out.push(pool[Math.floor(Math.random() * pool.length)]);
   124→  }
   125→  return out;
   126→};

   351→function normalizeStringArray(value: unknown): string[] {
   352→  if (Array.isArray(value)) {
   353→    return value
   354→      .filter((item): item is string => typeof item === "string")
   355→      .map((item) => item.trim())
   356→      .filter(Boolean);
   357→  }
   358→  if (typeof value === "string" && value.trim()) {
   359→    return [value.trim()];
   360→  }
   361→  return [];

   391→function normalizePersonaForProfile(persona: unknown, profile: BaseProfile): Persona {
   392→  const record = isRecord(persona) ? persona : {};

   394→  const normalized: Persona = {

   397→    mbti: profile.mbti,
   398→    gender: profile.gender,
   399→    age: profile.age,
   400→    voiceId: typeof record.voiceId === "string" && record.voiceId.trim() ? record.voiceId.trim() : undefined,
   401→    relationships: undefined,
   402→    basicInfo: profile.basicInfo,
   403→  };
   404→
   405→  for (const field of PERSONA_TEXT_FIELDS) {
   406→    const value = record[field];
   407→    if (typeof value === "string" && value.trim()) {
   408→      normalized[field] = value.trim();
   409→    }
   410→  }
   411→
   412→  for (const field of OPTIONAL_PERSONA_STRING_FIELDS) {
   413→    const value = record[field];
   414→    if (typeof value === "string" && value.trim()) {
   415→      normalized[field] = value.trim();
   416→    }
   417→  }
   418→
   419→  const triggerTopics = normalizeStringArray(record.triggerTopics);
   420→  if (triggerTopics.length > 0) {
   421→    normalized.triggerTopics = triggerTopics;
   422→  }
   423→
   424→  return normalized;

   441→const alignCharactersToProfiles = (
   442→  chars: unknown,
   443→  profiles: BaseProfile[]
   444→): GeneratedCharacter[] | null => {
   445→  if (!Array.isArray(chars)) {
   446→    console.error("[alignCharacters] chars is not an array:", chars);
   447→    return null;
   448→  }
   449→  if (chars.length !== profiles.length) {
   450→    console.error(`[alignCharacters] length mismatch: ${chars.length} chars vs ${profiles.length} profiles`);
   451→    return null;
   452→  }
   453→  const byName = new Map<string, GeneratedCharacter>();
   454→  for (const c of chars as GeneratedCharacter[]) {
   455→    if (!c || typeof c !== "object") {
   456→      console.error("[alignCharacters] invalid character object:", c);
   457→      return null;
   458→    }
   459→    const name = typeof c.displayName === "string" ? c.displayName.trim() : "";
   460→    if (!name) {
   461→      console.error("[alignCharacters] missing displayName:", c);
   462→      return null;
   463→    }
   464→    if (byName.has(name)) {
   465→      console.error("[alignCharacters] duplicate name:", name);
   466→      return null;
   467→    }
   468→    byName.set(name, c);
   469→  }
   470→  const ordered: GeneratedCharacter[] = [];
   471→  for (const profile of profiles) {
   472→    const key = profile.displayName.trim();
   473→    const rawCharacter = byName.get(key);
   474→    if (!rawCharacter) {
   475→      console.error(`[alignCharacters] character not found for profile: ${key}, available names:`, Array.from(byName.keys()));
   476→      return null;
   477→    }
   478→    const c = normalizeGeneratedCharacterForProfile(rawCharacter, profile);
   479→    if (!c || !isValidPersonaForProfile(c.persona, profile) || !isValidPlayerMind(c.playerMind)) {
   480→      const p = isRecord(rawCharacter) ? rawCharacter.persona : undefined;
   481→      console.error(`[alignCharacters] invalid persona for ${key}:`, {
   482→        rawCharacter,
   483→        normalizedCharacter: c,
   484→        profile: { gender: profile.gender, age: profile.age, mbti: profile.mbti },
   485→        isValid: c ? isValidPersona(c.persona) : false,
   486→        isValidPlayerMind: c ? isValidPlayerMind(c.playerMind) : false,
   487→        genderMatch: p?.gender === profile.gender,
   488→        ageMatch: p?.age === profile.age,
   489→        mbtiMatch: isRecord(p) ? String(p.mbti || "").trim() === profile.mbti : false,
   490→      });
   491→      return null;
   492→    }
   493→    ordered.push(c);
   494→  }
   495→  return ordered;
   496→};

   539→例如：werewolfExperience 写"信息博弈视角"而非"这个人认为狼人杀是信息博弈关键在于控制信息流"
   540→
   541→【persona 字段】（只生成以下 4 个）
   542→- werewolfExperience：短语，怎么理解狼人杀
   543→- vocabularyStyle：短语，常用词和术语密度
   544→- reasoningStyle：短语，最先注意什么
   545→- wolfDeceptionStyle：短语，拿狼时怎么伪装
   546→
   547→【playerMind 字段】（只生成以下 3 个）
   548→- courage：短语，敢不敢诈/硬推
   549→- suspicionThreshold：短语，几条线索改站边
   550→- selfProtection：短语，被攻击时怎么应对
   551→
   552→【voiceRules】每项 2-4 字短语，1-2 个，如 ["冷静","数据流"]
   553→
   554→【输出格式】
   555→输出 JSON 数组（共 ${count} 个），每个对象严格匹配以下 schema：
   556→[{"displayName":"中文名2-3字","gender":"male或female","age":25,"mbti":"INTJ","basicInfo":"一句话职业","persona":{"voiceRules":["短语"],"werewolfExperience":"短语","vocabularyStyle":"短语","reasoningStyle":"短语","wolfDeceptionStyle":"短语"},"playerMind":{"courage":"短语","suspicionThreshold":"短语","selfProtection":"短语"}}]
   557→
   558→【重要】
   559→- 名字各不相同，符合场景名字风格
   560→- 年龄 20-55，mbti 是 4 字母
   561→- 只输出 JSON 数组，不要 markdown 代码块、不要解释文字
   562→- 每个字段一句话短语，不要长篇大论`;
   563→  }
   564→  return `You are a character designer for a Werewolf game.
   565→
   566→[Scene]
   567→${scenario.title} - ${scenario.description}
   568→Role hints: ${scenario.rolesHint}
   569→Note: scene only flavors identity and names; gameplay stays on Werewolf itself.
   570→
   571→[Task]
   572→Generate ${count} player characters, each with base info and full persona. These are "ordinary people playing Werewolf", not mystery characters.
   573→
   574→[IMPORTANT: field values must be SHORT PHRASES (3-10 words), NOT full sentences]
   575→e.g. werewolfExperience: "info-game perspective" NOT "this person thinks Werewolf is about controlling information flow"
   576→
   577→[persona fields] (generate ONLY these 4)
   578→- werewolfExperience: short phrase, how they view Werewolf
   579→- vocabularyStyle: short phrase, jargon density
   580→- reasoningStyle: short phrase, what they notice first
   581→- wolfDeceptionStyle: short phrase, wolf disguise approach
   582→
   583→[playerMind fields] (generate ONLY these 3)
   584→- courage: short phrase, bluff/push willingness
   585→- suspicionThreshold: short phrase, clues needed to switch
   586→- selfProtection: short phrase, defense when attacked
   587→
   588→[voiceRules] each 2-4 word phrase, 1-2 items, e.g. ["calm","analytical"]
   589→
   590→[Output format]
   591→JSON array (${count} objects), each matching this schema:
   592→[{"displayName":"Name","gender":"male or female","age":25,"mbti":"INTJ","basicInfo":"one-line job","persona":{"voiceRules":["phrase"],"werewolfExperience":"phrase","vocabularyStyle":"phrase","reasoningStyle":"phrase","wolfDeceptionStyle":"phrase"},"playerMind":{"courage":"phrase","suspicionThreshold":"phrase","selfProtection":"phrase"}}]
   593→
   594→[Important]
   595→- Unique names, age 20-55, mbti 4 letters
   596→- Output ONLY the JSON array, no markdown, no explanation
   597→- One short phrase per field, keep it concise`;
   598→};
   599→
   600→export async function generateCharacters(
   601→  count: number,
   602→  scenario?: GameScenario,
   603→  options?: {
   604→    onBaseProfiles?: (profiles: BaseProfile[]) => void;
   605→    onCharacter?: (index: number, character: GeneratedCharacter) => void;
   606→  }
   607→): Promise<GeneratedCharacter[]> {
   608→  const usedScenario = scenario ?? getRandomScenario();
   609→  const runOnce = async () => {
   610→    const startTime = Date.now();
   611→    // 合并两阶段：一次调用生成完整角色（base info + persona + playerMind），
   612→    // 省掉单独的 base profiles 调用（~7s），总耗时减半。
   613→    const mergedPrompt = buildMergedPrompt(count, usedScenario);
   614→
   615→    // 流式生成：每解析出一个完整角色就立即回调
   616→    const finalizedCharacters: GeneratedCharacter[] = [];
   617→    const emittedIndices = new Set<number>();
   618→    const emittedBaseProfiles: BaseProfile[] = [];
   619→    let accumulatedContent = "";
   620→
   621→    // 精简字段后每角色约 400 tokens
   622→    const fullMaxTokens = Math.max(3500, count * 450 + 600);
   623→
   624→    const stream = generateCompletionStream({
   625→      model: getGeneratorModel(),
   626→      messages: [{ role: "user", content: mergedPrompt }],
   627→      temperature: GAME_TEMPERATURE.CHARACTER_GENERATION,
   628→      max_tokens: fullMaxTokens,
   629→      reasoning: CHARACTER_GENERATOR_REASONING,
   630→    });
   631→
   632→    for await (const chunk of stream) {
   633→      accumulatedContent += chunk;
   634→
   635→      // 提取完整的角色对象（含 displayName + persona + playerMind + base info）
   636→      const cleaned = stripMarkdownCodeFences(accumulatedContent);
   637→      const characterPattern = /\{\s*"displayName"\s*:\s*"[^"]+"\s*,\s*"gender"\s*:\s*"[^"]+"\s*,\s*"age"\s*:\s*\d+[^}]*"persona"\s*:\s*\{[^{}]*(?:\{[^{}]*\}[^{}]*)*\}\s*,\s*"playerMind"\s*:\s*\{[^{}]*(?:\{[^{}]*\}[^{}]*)*\}\s*\}/g;
   638→      const matches = cleaned.match(characterPattern);
   639→
   640→      if (matches) {
   641→        for (const match of matches) {
   642→          try {
   643→            const c = parseLLMJson<GeneratedCharacter & { gender?: string; age?: number; mbti?: string; basicInfo?: string }>(match);
   644→            if (!c || !c.displayName) continue;
   645→
   646→            // 用 displayName 去重（合并模式下没有预生成的 baseProfiles 数组）
   647→            const existingIndex = finalizedCharacters.findIndex(
   648→              (fc) => fc?.displayName === c.displayName
   649→            );
   650→            if (existingIndex !== -1) continue; // 已处理
   651→
   652→            const index = emittedBaseProfiles.length;
   653→            emittedBaseProfiles.push({
   654→              displayName: c.displayName,
   655→              gender: (c.gender === "female" ? "female" : "male") as "male" | "female",
   656→              age: typeof c.age === "number" ? c.age : 25,
   657→              mbti: c.mbti || "INTJ",
   658→              basicInfo: c.basicInfo || "",
   659→            });
   660→
   661→            // 名字一出就触发 baseProfiles 回调（让前端先显示名字）
   662→            options?.onBaseProfiles?.([...emittedBaseProfiles]);
   663→
   664→            // 构建完整角色
   665→            const persona = c.persona;
   666→            if (!persona || !Array.isArray(persona.voiceRules) || persona.voiceRules.length === 0) continue;
   667→
   668→            const voiceId = resolveVoiceId(
   669→              persona.voiceId,
   670→              persona.gender,
   671→              persona.age,
   672→              "zh" as AppLocale
   673→            );
   674→
   675→            const character: GeneratedCharacter = {
   676→              displayName: c.displayName,
   677→              persona: {
   678→                ...persona,
   679→                gender: persona.gender || (emittedBaseProfiles[index].gender as "male" | "female" | "nonbinary"),
   680→                age: persona.age || emittedBaseProfiles[index].age,
   681→                mbti: persona.mbti || emittedBaseProfiles[index].mbti,
   682→                basicInfo: emittedBaseProfiles[index].basicInfo,
   683→                voiceId,
   684→                relationships: undefined,
   685→              },
   686→              playerMind: c.playerMind,
   687→            };
   688→
   689→            finalizedCharacters[index] = character;
   690→            options?.onCharacter?.(index, character);
   691→            console.log(`[character-gen] emitted character ${index}: ${character.displayName}`);
   692→          } catch {
   693→            // 解析失败是正常的
   694→          }
   695→        }
   696→      }
   697→    }
   698→
   699→    // 流式结束后，检查是否所有角色都已生成
   700→    if (finalizedCharacters.filter(Boolean).length < count) {
   701→      // 回退到完整解析
   702→      const cleaned = stripMarkdownCodeFences(accumulatedContent);
   703→      const fullResult = parseLLMJson<unknown>(cleaned);
   704→      if (!fullResult) {
   705→        throw new Error("Character generation returned invalid JSON");
   706→      }
   707→
   708→      const normalized = normalizeGeneratedCharacters(fullResult);
   709→      const chars = normalized.characters;
   710→
   711→      // 补充未生成的角色（合并模式下无 baseProfiles，直接用生成结果）
   712→      for (let i = 0; i < chars.length; i++) {
   713→        if (finalizedCharacters[i]) continue;
   714→
   715→        const c = chars[i];
   716→        if (!c?.persona) continue;
   717→
   718→        const voiceId = resolveVoiceId(
   719→          c.persona.voiceId,
   720→          c.persona.gender,
   721→          c.persona.age,
   722→          "zh" as AppLocale
   723→        );
   724→
   725→        const character: GeneratedCharacter = {
   726→          displayName: c.displayName,
   727→          persona: {
   728→            ...c.persona,
   729→            voiceId,
   730→            relationships: undefined,
   731→          },
   732→          playerMind: c.playerMind,
   733→        };
   734→
   735→        finalizedCharacters[i] = character;
   736→        options?.onCharacter?.(i, character);
   737→      }
   738→    }
   739→
   740→    await aiLogger.log({
   741→      type: "character_generation",
   742→      request: { 
   743→        model: getGeneratorModel(),
   744→        messages: [{ role: "user", content: mergedPrompt }],
   745→      },
   746→      response: { 
   747→        content: JSON.stringify(finalizedCharacters.map((c) => ({
   748→          displayName: c.displayName,
   749→          hiddenCommunicationProfile: {
   750→            werewolfExperience: c.persona.werewolfExperience,
   751→            vocabularyStyle: c.persona.vocabularyStyle,
   752→            reasoningStyle: c.persona.reasoningStyle,
   753→            speechLengthHabit: c.persona.speechLengthHabit,
   754→            pressureStyle: c.persona.pressureStyle,
   755→            uncertaintyStyle: c.persona.uncertaintyStyle,
   756→            mistakePattern: c.persona.mistakePattern,
   757→            wolfDeceptionStyle: c.persona.wolfDeceptionStyle,
   758→          },
   759→          playerMind: c.playerMind,
   760→        }))),
   761→        duration: Date.now() - startTime 
   762→      },
   763→    });
   764→
   765→    return finalizedCharacters;
   766→  };
   767→
   768→  let lastError: unknown;
   769→  for (let attempt = 0; attempt < 2; attempt += 1) {
   770→    try {
   771→      console.log(
   772→        `[character-gen] Attempt ${attempt + 1}/2, customKeyEnabled: ${isCustomKeyEnabled()}`
   773→      );
   774→      return await runOnce();
   775→    } catch (error) {
   776→      lastError = error;
   777→      console.error(`[character-gen] Attempt ${attempt + 1} failed:`, error);
   778→
   779→      const errorMsg = String(error);
   780→      const isQuotaError = errorMsg.includes("[QUOTA_EXHAUSTED]") ||
   781→                          errorMsg.includes("402") ||
   782→                          errorMsg.includes("insufficient") ||
   783→                          errorMsg.includes("余额");
   784→
   785→      // Quota exhaustion is not recoverable by retrying — abort immediately
   786→      // (applies to both built-in z.ai quota and custom-key quota).
   787→      if (isQuotaError) {
   788→        console.error("[character-gen] Quota exhausted, aborting retry");
   789→        throw error;
   790→      }
   791→      
   792→      if (attempt === 0) {
   793→        continue;
   794→      }
   795→      console.error("Character generation failed:", error);
   796→      await aiLogger.log({
   797→        type: "character_generation",
   798→        request: { 
   799→          model: GENERATOR_MODEL,
   800→          messages: [{ role: "user", content: "(two-stage generation)" }],
   801→        },
   802→        response: { content: "[]", duration: 0 },
   803→        error: String(error),
   804→      });
   805→    }
   806→  }
   807→
   808→  throw lastError;
   809→}
