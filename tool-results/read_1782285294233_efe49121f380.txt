   102→function getRitualCueFromSystemMessage(content: string): { title: string; subtitle?: string } | null {
   103→  const { t } = getI18n();
   104→  const systemMessages = getSystemMessages();
   105→  const patterns = getSystemPatterns();
   106→  const nightFallRegex = new RegExp(patterns.nightFall);
   107→  const playerKilledRegex = new RegExp(patterns.playerKilled);
   108→  const playerPoisonedRegex = new RegExp(patterns.playerPoisoned);
   109→  const badgeElectedRegex = new RegExp(patterns.badgeElected);
   110→  const playerExecutedRegex = new RegExp(patterns.playerExecuted);
   111→  const text = content.trim();
   112→  if (text === systemMessages.gameStart) return { title: t("ritual.gameStart") };
   113→  if (nightFallRegex.test(text)) return { title: text };
   114→  if (text === systemMessages.guardActionStart) return { title: text };
   115→  if (text === systemMessages.wolfActionStart) return { title: text };
   116→  if (text === systemMessages.witchActionStart) return { title: text };
   117→  if (text === systemMessages.seerActionStart) return { title: text };
   118→  if (text === systemMessages.peacefulNight) return { title: systemMessages.peacefulNight };
   119→  if (playerKilledRegex.test(text)) return { title: text };
   120→  if (playerPoisonedRegex.test(text)) return { title: text };
   121→  if (text === systemMessages.dayBreak) return { title: systemMessages.dayBreak };
   122→  if (text === t("badgePhase.signupStart")) return { title: t("ritual.badgeSignup") };
   123→  if (text === systemMessages.badgeSpeechStart) {
   124→    return { title: t("ritual.badgeSpeechTitle"), subtitle: t("ritual.badgeSpeechSubtitle") };
   125→  }
   126→  if (text === systemMessages.badgeElectionStart) return { title: systemMessages.badgeElectionStart };
   127→  if (text === systemMessages.badgeRevote) return { title: systemMessages.badgeRevote };
   128→  if (badgeElectedRegex.test(text)) return { title: text };
   129→  if (text === systemMessages.dayDiscussion) return { title: systemMessages.dayDiscussion };
   130→  if (text === systemMessages.voteStart) return { title: systemMessages.voteStart };
   131→  if (playerExecutedRegex.test(text)) return { title: text };
   132→  if (text === systemMessages.voteTie) return { title: systemMessages.voteTie };
   133→  return null;
   134→}
