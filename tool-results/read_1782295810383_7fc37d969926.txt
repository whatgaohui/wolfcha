   102→   102→// Flatten multipart content to plain string for models that don't support it
   103→   103→function flattenMultipartContent(messages: unknown[]): unknown[] {
   104→   104→  if (!Array.isArray(messages)) return messages;
   105→   105→
   106→   106→  return messages.map((msg) => {
   107→   107→    if (!msg || typeof msg !== "object") return msg;
   108→   108→    const m = msg as Record<string, unknown>;
   109→   109→
   110→   110→    // If content is an array, flatten to string
   111→   111→    if (Array.isArray(m.content)) {
   112→   112→      const textParts = m.content
   113→   113→        .filter((part): part is { type: string; text: string } =>
   114→   114→          part && typeof part === "object" && (part as { type?: string }).type === "text"
   115→   115→        )
   116→   116→        .map((part) => part.text || "")
   117→   117→        .filter(Boolean);
   118→   118→      
   119→   119→      return { ...m, content: textParts.join("\n\n") };
   120→   120→    }
   121→   121→
   122→   122→    return m;
   123→   123→  });
   124→   124→}

   128→   128→function coalesceTextOnlyMultipartContent(messages: unknown[]): unknown[] {
   129→   129→  if (!Array.isArray(messages)) return messages;
   130→   130→
   131→   131→  return messages.map((msg) => {
   132→   132→    if (!msg || typeof msg !== "object") return msg;

   158→   158→function prependDeepSeekStablePrefix(messages: unknown[]): unknown[] {
   159→   159→  if (!Array.isArray(messages)) return messages;
   160→   160→  let prepended = false;

   162→   162→    if (prepended || !msg || typeof msg !== "object") return msg;
   163→   163→    const m = msg as Record<string, unknown>;
   164→   164→    if (m.role !== "system" || typeof m.content !== "string") return m;
   165→   165→    prepended = true;
   166→   166→    if (m.content.includes(DEEPSEEK_STABLE_PREFIX_MARKER)) return m;
   167→   167→    return {
   168→   168→      ...m,
   169→   169→      content: `${DEEPSEEK_STABLE_PROMPT_CACHE_PREFIX}\n\n${m.content}`,
   170→   170→    };
   171→   171→  });
   172→   172→
   173→   173→  if (prepended) return next;
   174→   174→  return [
   175→   175→    { role: "system", content: DEEPSEEK_STABLE_PROMPT_CACHE_PREFIX },
   176→   176→    ...next,
   177→   177→  ];

   292→   292→async function runBatchItem(
   293→   293→  payload: ChatRequestPayload,
   294→   294→  headerApiKey: string | null,
   295→   295→  headerDashscopeKey: string | null,

   301→   301→    messages,

   331→   331→    }
   332→   332→  }

   338→   338→    modelRefOverride?.temperature !== undefined
   339→   339→      ? modelRefOverride.temperature
   340→   340→      : (typeof temperature === "number" && Number.isFinite(temperature) ? temperature : 0.7);
   341→   341→  const cappedTemperature = (() => {
   342→   342→    const lower = typeof model === "string" ? model.toLowerCase() : "";

   344→   344→      modelProvider === "zenmux" ||
   345→   345→      lower.startsWith("moonshotai/") ||
   346→   346→      lower.includes("kimi");

   348→   348→      return Math.min(Math.max(0, normalizedTemperature), 1);
   349→   349→    }
   350→   350→    return Math.max(0, normalizedTemperature);
   351→   351→  })();

   353→   353→

   358→   358→    processedMessages = prependDeepSeekStablePrefix(coalesceTextOnlyMultipartContent(processedMessages));
   359→   359→  } else if (modelProvider === "dashscope") {
   360→   360→    processedMessages = stripCacheControl(processedMessages);
   361→   361→  } else if (!supportsExplicitCaching(model)) {
   362→   362→    processedMessages = stripCacheControl(processedMessages);
   363→   363→  }

   372→   372→    }

   374→   374→    const normalizedModel = normalizeDashscopeModelName(model);
   375→   375→    const normalizedResponseFormat = response_format as { type?: unknown } | undefined;
   376→   376→    const dashscopeMessages =
   377→   377→      normalizedResponseFormat?.type === "json_object"
   378→   378→        ? withDashscopeJsonHint(processedMessages)
   379→   379→        : processedMessages;
   380→   380→
   381→   381→    const requestBody: Record<string, unknown> = {
   382→   382→      model: normalizedModel,
   383→   383→      messages: dashscopeMessages,
   384→   384→      temperature: cappedTemperature,
   385→   385→    };
   386→   386→
   387→   387→    if (typeof max_tokens === "number" && Number.isFinite(max_tokens)) {
   388→   388→      requestBody.max_tokens = Math.max(16, Math.floor(max_tokens));
   389→   389→    }
   390→   390→
   391→   391→    if (response_format) {
   392→   392→      requestBody.response_format = response_format;
   393→   393→    }
   394→   394→
   395→   395→    const controller = new AbortController();
   396→   396→    const timeoutId = setTimeout(() => controller.abort(), API_TIMEOUT_MS);

   399→   399→    try {
   400→   400→      response = await fetch(DASHSCOPE_CHAT_COMPLETIONS_URL, {
   401→   401→        method: "POST",
   402→   402→        headers: {
   403→   403→          Authorization: `Bearer ${dashscopeApiKey}`,
   404→   404→          "Content-Type": "application/json",
   405→   405→        },
   406→   406→        body: JSON.stringify(requestBody),
   407→   407→        signal: controller.signal,
   408→   408→      });
   409→   409→    } finally {
   410→   410→      clearTimeout(timeoutId);
   411→   411→    }
   412→   412→
   413→   413→    if (!response.ok) {
   414→   414→      const errorText = await response.text();
   415→   415→      let parsed: unknown = undefined;
   416→   416→      try {
   417→   417→        parsed = JSON.parse(errorText);
   418→   418→      } catch {
   419→   419→        // ignore
   420→   420→      }
   421→   421→      return {
   422→   422→        ok: false,
   423→   423→        status: response.status,
   424→   424→        error: `DashScope API error: ${response.status}`,
   425→   425→        details: parsed ?? errorText,
   426→   426→      };
   427→   427→    }
   428→   428→
   429→   429→    const result = await response.json();
   430→   430→    return { ok: true, data: result };
   431→   431→  }
   432→   432→
   433→   433→  if (modelProvider === "tokendance") {
   434→   434→    if (hasAnyCustomKeyHeader && !headerTokendanceKey) {
   435→   435→      return { ok: false, status: 401, error: "已启用自定义 Key，但未提供 TokenDance Key（已拒绝回退到系统 Key）" };
